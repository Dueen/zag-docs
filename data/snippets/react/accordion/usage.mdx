```jsx
import * as accordion from "@ui-machines/accordion"
import { useMachine } from "@ui-machines/react"
import { useMemo } from "react"

const data = [
  { title: "Watercraft", content: "Sample accordion content" },
  { title: "Automobiles", content: "Sample accordion content" },
  { title: "Aircrafts", content: "Sample accordion content" },
]

function Accordion() {
  const [state, send] = useMachine(accordion.machine)
  const api = useMemo(() => Accordion.connect(state, send), [state])

  return (
    <div {...api.rootProps}>
      {data.map((item) => (
        <div {...api.getItemProps({ value: item.value })}>
          <h3>
            <button {...api.getTriggerProps({ value: item.value })}>
              {item.title}
            </button>
          </h3>
          <div {...api.getContentProps({ value: item.value })}>
            {item.content}
          </div>
        </div>
      ))}
    </div>
  )
}
```
