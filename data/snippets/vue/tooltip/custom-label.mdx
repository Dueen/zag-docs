import * as tooltip from "@ui-machines/tooltip"
import { normalizeProps, useMachine, useSetup } from "@ui-machines/vue"
import { computed, defineComponent, h, Fragment } from "vue"

export default defineComponent({
  name: "Tooltip",
  setup() {
    const [state, send] = useMachine(
      tooltip.machine.withContext({
        "aria-label": "Tooltip Custom Label",
      }),
    )
    const ref = useSetup({ send, id: "1" })
    const apiRef = computed(() =>
      tooltip.connect(state.value, send, normalizeProps),
    )

    return () => {
      const api = apiRef.current
      return (
        <>
          <div>
            <button ref={ref} {...api.triggerProps}>
              Hover me
            </button>
            {api.isOpen && (
              <div {...api.positionerProps}>
                <div {...api.contentProps}>
                  {/* rendered */}
                  <span>Tooltip</span>
                  {/* announced */}
                  <div {...api.labelProps} />
                </div>
              </div>
            )}
          </div>
        </>
      )
    }
  },
})
