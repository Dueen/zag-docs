```jsx
import * as rangeSlider from "@ui-machines/range-slider"
import { normalizeProps, useMachine, useSetup } from "@ui-machines/vue"
import { defineComponent, computed, h, Fragment } from "vue"

export default defineComponent({
  name: "RangeSlider",
  setup() {
    const [state, send] = useMachine(
      rangeSlider.machine.withContext({
        name: ["min", "max"],
        value: [10, 60],
      }),
    )
    const ref = useSetup({ send, id: "1" })
    const apiRef = computed(() =>
      rangeSlider.connect(state.value, send, normalizeProps),
    )

    return () => {
      const api = apiRef.value
      return (
        <div ref={ref} {...api.rootProps}>
          <div {...api.trackProps}>
            <div {...api.rangeProps} />
          </div>
          {values.map((_, index) => (
            <div key={index} {...api.getThumbProps(index)}>
              <input {...api.getInputProps(index)} />
            </div>
          ))}
        </div>
      )
    }
  },
})
```
