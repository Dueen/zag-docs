```jsx
import * as slider from "@ui-machines/slider"
import { normalizeProps, useMachine, useSetup } from "@ui-machines/vue"
import { defineComponent, computed, h, Fragment } from "vue"

export default defineComponent({
  name: "Slider",
  setup() {
    const [state, send] = useMachine(slider.machine)
    const ref = useSetup({ send, id: "1" })
    const apiRef = computed(() =>
      slider.connect(state.value, send, normalizeProps),
    )

    return () => {
      const api = apiRef.value
      return (
        <div ref={ref}>
          <div>
            <label {...api.labelProps}>Slider Label</label>
            <output {...api.outputProps}>{api.value}</output>
          </div>
          <div {...api.rootProps}>
            <div {...api.trackProps}>
              <div {...api.rangeProps} />
            </div>
            <div {...api.thumbProps}>
              <input {...api.inputProps} />
            </div>
          </div>
        </div>
      )
    }
  },
})
```
